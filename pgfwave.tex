\makeatletter
\def\textwaveX#1{%
  \ifx\relax#1\empty
  \else
    \def\currentchar{#1}%
    \@ifundefined{path\currentchar}{%
      \edef\str{\str\csname path\lastchar\currentchar\endcsname}%
      \let\lastchar\currentchar
    }{%
      \edef\str{\str\csname path\currentchar\endcsname}%
    }
    \expandafter
    \textwaveX
  \fi
}
