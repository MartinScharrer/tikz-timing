\def\pgfwavesetslope#1{%
  \@tempcnta=#1\relax

  \edef\nslope{\the\@tempcnta}%

\def\textwaveX#1{%
  \ifx\relax#1\empty
  \else
    \def\currentchar{#1}%
    \edef\str{\str\csname path\lastchar\currentchar\endcsname}%
    \let\lastchar\currentchar
    \expandafter
    \textwaveX
  \fi
}
